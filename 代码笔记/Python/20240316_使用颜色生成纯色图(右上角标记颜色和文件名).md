from PIL import Image, ImageDraw, ImageFont

# 定义颜色列表
colors = ["#FF0000", "#FF3300", "#FF6600", "#FF9900", "#FFCC00", "#FFFF00", "#CCFF00", "#99FF00", "#66FF00", "#33FF00", "#00FF00", "#00FF33", "#00FF66", "#00FF99", "#00FFCC", "#00FFFF", "#00CCFF", "#0099FF", "#0066FF", "#0033FF", "#0000FF", "#3300FF", "#6600FF", "#9900FF", "#CC00FF", "#FF00FF", "#FF00CC", "#FF0099", "#FF0066", "#FF0033", "#FF0000"]

# 创建图片
def create_image(color, filename):
    # 创建一张空白图片
    img = Image.new('RGB', (1920, 1080), color)
    draw = ImageDraw.Draw(img)

    # 在右上角添加文字标注
    font = ImageFont.truetype("arial.ttf", 30)
    text = f"{color}\n{filename}"
    text_width, text_height = draw.textsize(text, font)
    draw.text((1920 - text_width - 20, 20), text, fill="black", font=font)

    # 保存图片
    img.save(filename + '.png')

# 生成图片
for i, color in enumerate(colors):
    create_image(color, f'color{i}')
