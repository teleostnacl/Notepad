``` python
import cv2
import os

# 设置要读取的图片文件夹路径
image_folder = 'path/to/your/images'

# 获取图片文件夹下所有图片的文件名，并按数字顺序排序
images = sorted([f"color{i}.png" for i in range(31)])

# 设置每张图片停留的帧数
frames_per_image = 90  # 假设帧率为30fps，则3秒 = 30帧/秒 * 3秒 = 90帧

# 初始化视频写入对象
img = cv2.imread(os.path.join(image_folder, images[0]))
height, width, layers = img.shape
video = cv2.VideoWriter('output.mp4', cv2.VideoWriter_fourcc(*'mp4v'), 30, (width, height))

# 逐个读取图片并写入视频
for image in images:
    img_path = os.path.join(image_folder, image)
    img = cv2.imread(img_path)
    for _ in range(frames_per_image):
        video.write(img)

# 释放视频对象
cv2.destroyAllWindows()
video.release()

print("视频生成完成！")
```
